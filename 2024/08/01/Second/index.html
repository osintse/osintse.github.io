正则表达式

只看了re.research,语言学习一定要多重复，不然很容易忘，我的C++是这样的orz。想到科研训练还要学MySQL,这几天一定学

#正则表达式

import re
re.search(r'AN', 'GUAN')
#通配符(.)
# \可以剥夺元字符的特殊能力，也可以使普通字符拥有特殊能力
re.search(r'\d\d\d\.\d\d\d\.\d\d\d\.\d\d\d', 'other112.187.244.253')

re.search(r'[aeiouAEIOU]', "GUaN")

re.search(r'[0-9][0-9][0-9]', 'I love 999')

#重复匹配
re.search(r'ab{3}c','abbbc')
re.search(r'ab{3,5}c', 'abbbbbc')

#匹配0~255的数字
re.search(r'[0-1]\d\d|2[0-4]\d|25[0-5]', '188')

#匹配ip地址
re.search(r'(([01]{0,1}\d{0,1}\d|2[0-4]\d|25[0-5])\.){3}([01]{0,1}\d{0,1}\d|2[0-4]\d|25[0-5])', 'other192.168.1.1other')
